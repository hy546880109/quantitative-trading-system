### 3.2 策略层实现

#### 技术指标计算引擎

```python
class TechnicalIndicators:
    """技术指标计算类"""

    def __init__(self, data: pd.DataFrame):
        self.data = data.copy()

    def sma(self, window: int = 10) -> pd.Series:
        """简单移动平均线"""
        return self.data['close'].rolling(window=window).mean()

    def rsi(self, window: int = 14) -> pd.Series:
        """相对强弱指数"""
        delta = self.data['close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def macd(self, fast: int = 12, slow: int = 26, signal: int = 9):
        """MACD指标"""
        exp1 = self.data['close'].ewm(span=fast).mean()
        exp2 = self.data['close'].ewm(span=slow).mean()
        macd_line = exp1 - exp2
        signal_line = macd_line.ewm(span=signal).mean()
        histogram = macd_line - signal_line
        return macd_line, signal_line, histogram
```

#### 移动平均策略示例

```python
class MovingAverageStrategy:
    """移动平均线策略"""

    def __init__(self, short_window: int = 20, long_window: int = 50):
        self.short_window = short_window
        self.long_window = long_window

    def generate_signals(self, data: pd.DataFrame) -> pd.Series:
        """生成交易信号"""
        ti = TechnicalIndicators(data)
        short_ma = ti.sma(window=self.short_window)
        long_ma = ti.sma(window=self.long_window)

        signals = pd.Series(0, index=data.index)
        signals[short_ma > long_ma] = 1   # 买入信号
        signals[short_ma < long_ma] = -1  # 卖出信号

        return signals
```

### 3.3 回测引擎设计

回测引擎是量化交易系统的核心，负责模拟历史交易并计算绩效指标：

```python
class BacktestEngine:
    """回测引擎类"""

    def __init__(self, initial_capital: float = 10000):
        self.initial_capital = initial_capital
        self.cash = initial_capital
        self.portfolio = {}
        self.trades = []
        self.equity_curve = []

    def run_backtest(self, strategy, data: pd.DataFrame, symbol: str = "TEST"):
        """运行回测"""
        self.reset()

        # 生成交易信号
        signals = strategy.generate_signals(data)

        # 逐日执行交易
        for i, (idx, row) in enumerate(data.iterrows()):
            current_signal = signals.iloc[i]
            current_price = row['close']

            self.execute_trade(symbol, current_signal, current_price, str(idx))

            # 记录资金曲线
            portfolio_value = self.calculate_portfolio_value({symbol: current_price})
            self.equity_curve.append({
                'date': str(idx),
                'portfolio_value': portfolio_value,
                'cash': self.cash
            })

        return self._calculate_performance_metrics()

    def _calculate_performance_metrics(self):
        """计算绩效指标"""
        final_value = self.calculate_portfolio_value({})
        total_return = (final_value - self.initial_capital) / self.initial_capital

        returns = self._calculate_returns()
        sharpe_ratio = self._calculate_sharpe_ratio(returns)
        max_drawdown = self._calculate_max_drawdown()

        return {
            'total_return': total_return,
            'sharpe_ratio': sharpe_ratio,
            'max_drawdown': max_drawdown,
            'trades': self.trades,
            'equity_curve': self.equity_curve
        }
```
